    \begin{indt}{\section{Bornes inférieures, supérieures}}
        
        Soit $A \in \mathcal P(\R) \setminus \varnothing.$
        
        (1) Soit $M$ un majorant de $A$, \textit{i.e}, $\forall x \in A, x \le M$. On a alors :
            \[ M = \sup(A) \ssi \exists (u_n)_{n \in \N} \subset A\ |\ u_n \tendsto{n \to +\infty} M \]
        
        
        (2) Soit $m$ un minorant de $A$, \textit{i.e}, $\forall x \in A, m \le x$. On a alors :
            \[ m = \inf(A) \ssi \exists (u_n)_{n \in \N} \subset A\ |\ u_n \tendsto{n \to +\infty} m \]
        
    \end{indt}
    
    \vspace{12pt}
    
    \begin{indt}{\section{Densité}}
        
        Soit $A \in \mathcal P(\R)$. Alors $A$ est \textit{dense} dans $\R$ si, et seulement si :
            \[ \forall x \in \R,\ \exists (u_n)_{n \in \N} \subset A\ |\ u_n \tendsto{n \to +\infty} x \]
        
    \end{indt}
    
    \vspace{12pt}
    
    \begin{indt}{\section{Limites}}
        
        Soient $a, l \in \cj \R$, et $f$ une fonction définie au voisinage de $a$. Alors :
            \[
                f(x) \tendsto{x \to a} l
                \ssi \forall (u_n)_{n \in \N} \subset \mathcal D_f\ |\ u_n \tendsto{n \to +\infty} a,\
                f(u_n) \tendsto{n \to +\infty} l
            \]
        
    \end{indt}
    
    \vspace{12pt}
    
    \begin{indt}{\section{Continuité}}
        
        Soit $a \in \R$, et $f$ une fonction définie en $a$. Alors :
            \[
                f(x) \tendsto{x \to a} f(a)
                \ssi \forall (u_n)_{n \in \N} \subset \mathcal D_f\ |\ u_n \tendsto{n \to +\infty} a,\
                f(u_n) \tendsto{n \to +\infty} f(a)
            \]
        
    \end{indt}
